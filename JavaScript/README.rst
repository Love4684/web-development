factorial of a number
===============================================================================

.. code:: javascript

    function factorial(n){
        //base case
        if(n == 0 || n == 1){
            return 1;
        //recursive case
        }else{
            return n * factorial(n-1);
        }
    }
    let n = 4;
    answer = factorial(n)
    console.log("The factorial of " + n + " is " + answer);
